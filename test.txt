#include <stdio.h>

int main() {
    FILE *fp;
    char c[100];

    fp = fopen("test.txt", "w");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }
    printf("Enter first line: ");
    fgets(c, 100, stdin);
    fputs(c, fp);
    fclose(fp);

    fp = fopen("test.txt", "a");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }
    printf("Enter second line: ");
    fgets(c, 100, stdin);
    fputs(c, fp);
    fclose(fp);

    fp = fopen("test.txt", "r");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }
    printf("\nFile contents:\n");
    while (fgets(c, 100, fp) != NULL) {
        printf("%s", c);
    }
    fclose(fp);

    return 0;
}
